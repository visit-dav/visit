MACRO(CREATE_EXEC name sources libraries)
  ADD_EXECUTABLE("${name}" ${sources})
  TARGET_LINK_LIBRARIES("${name}" ${libraries})
  INSTALL_TARGETS(/bin "${name}")
ENDMACRO(CREATE_EXEC)
